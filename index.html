import "std/io"
import "std/mem"

struct Token {
    kind: TokenKind,
    lexeme: ^char,
    length: i32,
    line: i32,
}

enum TokenKind {
    TK_IDENT,
    TK_NUMBER,
    TK_STRING,
    TK_LPAREN,
    TK_RPAREN,
    TK_LBRACE,
    TK_RBRACE,
    TK_SEMI,
    TK_EOF,
}

fn scan(src: ^char, len: i32) -> ^Token {
    let tokens = mem::alloc(Token, len)
    let pos: i32 = 0
    let line: i32 = 1

    while pos < len {
        let ch = src[pos]
        if ch == '\n' {
            line = line + 1
            pos = pos + 1
            continue
        }
        if is_alpha(ch) {
            let start = pos
            while pos < len and is_alnum(src[pos]) {
                pos = pos + 1
            }
            emit(tokens, TK_IDENT, src + start, pos - start, line)
        }
    }
    return tokens
}

fn emit(buf: ^Token, kind: TokenKind,
        lex: ^char, len: i32, line: i32) {
    buf.kind   = kind
    buf.lexeme = lex
    buf.length = len
    buf.line   = line
}

import "std/buf"
import "std/asm"

struct Emitter {
    code: ^buf::Buffer,
    data: ^buf::Buffer,
    stack_off: i32,
    label_id: i32,
}

fn emit_prologue(e: ^Emitter) {
    asm::push(e.code, asm::RBP)
    asm::mov(e.code, asm::RBP, asm::RSP)
    e.stack_off = 0
}

fn emit_epilogue(e: ^Emitter) {
    asm::mov(e.code, asm::RSP, asm::RBP)
    asm::pop(e.code, asm::RBP)
    asm::ret(e.code)
}

fn emit_call(e: ^Emitter, addr: u64) {
    asm::mov_imm(e.code, asm::RAX, addr)
    asm::call_reg(e.code, asm::RAX)
}

fn emit_binary(e: ^Emitter, op: BinOp,
               dst: Reg, src: Reg) {
    match op {
        BinOp::Add => asm::add(e.code, dst, src),
        BinOp::Sub => asm::sub(e.code, dst, src),
        BinOp::Mul => asm::imul(e.code, dst, src),
        BinOp::Div => {
            asm::mov(e.code, asm::RAX, dst)
            asm::cqo(e.code)
            asm::idiv(e.code, src)
            asm::mov(e.code, dst, asm::RAX)
        }
    }
}

fn new_label(e: ^Emitter) -> i32 {
    e.label_id = e.label_id + 1
    return e.label_id
}

import "std/io"

struct Parser {
    tokens: ^Token,
    current: i32,
    count: i32,
    had_error: bool,
}

fn parse_fn(p: ^Parser) -> ^ASTNode {
    expect(p, TK_FN)
    let name = expect(p, TK_IDENT)
    expect(p, TK_LPAREN)
    let params = parse_params(p)
    expect(p, TK_RPAREN)

    let ret_type: ^Type = null
    if match(p, TK_ARROW) {
        ret_type = parse_type(p)
    }

    let body = parse_block(p)
    return ast::fn_decl(name, params, ret_type, body)
}

fn parse_block(p: ^Parser) -> ^ASTNode {
    expect(p, TK_LBRACE)
    let stmts = vec::new(^ASTNode)

    while !check(p, TK_RBRACE) and !is_at_end(p) {
        vec::push(stmts, parse_stmt(p))
    }
    expect(p, TK_RBRACE)
    return ast::block(stmts)
}

fn parse_if(p: ^Parser) -> ^ASTNode {
    expect(p, TK_IF)
    let cond = parse_expr(p)
    let then = parse_block(p)
    let else_b: ^ASTNode = null
    if match(p, TK_ELSE) {
        else_b = parse_block(p)
    }
    return ast::if_stmt(cond, then, else_b)
}

fn expect(p: ^Parser, kind: TokenKind) -> Token {
    if !check(p, kind) {
        error(p, "unexpected token")
    }
    return advance(p)
}

import "std/mem"
import "std/fs"

struct PEHeader {
    magic: u16,
    sections: u16,
    timestamp: u32,
    entry_point: u32,
    image_base: u64,
    section_align: u32,
    file_align: u32,
}

struct Section {
    name: [char; 8],
    vsize: u32,
    vaddr: u32,
    raw_size: u32,
    raw_ptr: u32,
    flags: u32,
}

fn link(code: ^buf::Buffer,
        data: ^buf::Buffer,
        out: ^char) -> bool {
    let pe = build_header(code, data)
    let text = make_section(
        ".text", code,
        0x60000020
    )
    let rdata = make_section(
        ".rdata", data,
        0x40000040
    )
    defer mem::free(pe)
    return fs::write(out, pe)
}

fn align(val: u32, a: u32) -> u32 {
    return (val + a - 1) & ~(a - 1)
}
